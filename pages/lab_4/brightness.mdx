## Controlling LED brightness using PWM.

### Code
```
module pwm_brightness(
    input clk,           // Clock input
    input [7:0] duty,    // Duty cycle input
    output reg pwm_out,  // PWM output
    output reg LEDR1,
    output reg LEDR2,
    output reg LEDR3,
    output reg LEDR4,
    output reg LEDR5,
    output reg LEDR6,
    output reg LEDR7,
    output reg LEDR8,
    output reg LEDR9
);
    // Initialize all LEDs except LEDR0 to 'off'
    initial begin
        LEDR1 = 0;
        LEDR2 = 0;
        LEDR3 = 0;
        LEDR4 = 0;
        LEDR5 = 0;
        LEDR6 = 0;
        LEDR7 = 0;
        LEDR8 = 0;
        LEDR9 = 0;
    end

    reg [7:0] counter = 0;

    // PWM logic
    always @(posedge clk) begin
        counter <= counter + 1;
        pwm_out <= (counter < duty) ? 1'b1 : 1'b0;
    end
endmodule



```
